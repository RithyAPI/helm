1. Get Keycloak URL:

{{- if .Values.ingress.enabled }}
  https://{{ .Values.ingress.hostname }}{{ .Values.httpRelativePath }}
{{- else }}
  export POD=$(kubectl get pods -n {{ .Release.Namespace }} -l {{ include "keycloak.selectorLabels" . | replace "\n" "," }} -o jsonpath="{.items[0].metadata.name}")
  kubectl -n {{ .Release.Namespace }} port-forward pod/$POD 8080:8080
  Then open http://localhost:8080{{ .Values.httpRelativePath }}
{{- end }}

2. Admin credentials are stored in Secret: {{ include "keycloak.adminSecretName" . }}

3. Database:
{{- if .Values.postgresql.enabled }}
   Using bundled PostgreSQL Service: {{ include "keycloak.postgres.fullname" . }}:{{ .Values.externalDatabase.port }}
{{- else if .Values.externalDatabase.enabled }}
   Using external database at {{ .Values.externalDatabase.host }}:{{ .Values.externalDatabase.port }}
{{- else }}
   No database configured!
{{- end }}
